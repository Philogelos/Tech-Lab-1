<html>
      <canvas id="can" width="800" height="500" style="border: 11px solid black;"></canvas> 
      <script>
          let c = document.getElementById("can")
          let ctx = c.getContext("2d")
   
          let image = new Image()
          image.src = "explosion-spritesheet.png"
   
       let spriteSheetWidth = image.width
       let spriteSheetHeight = image.height
   
       let numberCols = 10
       let numberRows = 5
       let lastRowFrames = 7 //number of frames on last row
   
   
          let frameWidth = spriteSheetWidth/numberCols   //width of spritesheet divided by the number of columns
          let frameHeight = spriteSheetHeight/numberRows   //height of spritesheet divided by the number of rows
          let col = 0
          let row = 0
          let time = 0
   
       let frameDelay = 6 // how many screen-refresh frames to show each sprite-sheet frame
       
          function draw(){
              ctx.clearRect(0, 0, c.width, c.height)
              time++
              if(time > frameDelay){ 
                  col++
                  time = 0
              }
              if(col > numberCols-1){  //-1 since we start at col 0
                  col = 0
               row++
           }
           if(row == numberRows-1 && col == lastRowFrames){  
                  col = 0
               row = 0
           }
              
           ctx.drawImage(image, col*frameWidth, row*frameHeight, frameWidth, frameHeight,
                      0, 0, frameWidth, frameHeight)
   
              requestAnimationFrame(draw)
          }
          draw()  	
   
      </script>
   </html>
