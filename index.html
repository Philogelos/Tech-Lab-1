<html>
   <canvas id="can" width="800" height="500" style="border: 11px solid black;"></canvas> 
   <script>
   	let c = document.getElementById("can")
   	let ctx = c.getContext("2d")

   	let image = new Image()
   	image.src = "image-name.png"

   	let frameWidth = 745/5   //width of spritesheet divided by the number of columns
   	let frameHeight = 335/3   //height of spritesheet divided by the number of rows
   	let col = 0
   	let row = 0
   	let time = 0

   	function draw(){
       	ctx.clearRect(0, 0, c.width, c.height)
       	time++
       	if(time > 6){ //6 slows the frame change to 10 times a second - fps = 10
           	col++
           	time = 0
       	}
       	if(col > 4){  //this works only if we have a one row spritesheet
           	col = 0
		}
       	
ctx.drawImage(image, col*frameWidth, row*frameHeight, frameWidth, frameHeight,
           		0, 0, frameWidth, frameHeight)

       	requestAnimationFrame(draw)
   	}
   	draw()  	

   </script>
</html>
