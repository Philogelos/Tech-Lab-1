<html>
    <h1 id="header">Basic Template</h1>
    <p>In this basic game template we simply draw a red rectangle at (300, 100) of 150 pixels wide and 50 pixels high.
    <canvas id="can" width="1000" height="800" style="border: 5px solid black;"></canvas>  
  
    <script>
        let c = document.getElementById("can")
        let ctx = c.getContext("2d")
      
        function draw(){
            ctx.clearRect(0, 0, c.width, c.height)          
          
            ctx.fillStyle = "red"
            ctx.fillRect(300, 100, 150, 50)
           
            requestAnimationFrame(draw)
        }
        draw()
    </script>
</html>
